

select begdoc#, enddoc#, srm_begattach, srm_endattach, id, attachpid, attachlvl from tbldoc with(nolock)
where [path] like '%REV004%'
order by begdoc#

select a.begdoc#, b.begdoc#, b.enddoc#, b.srm_begattach, b.srm_endattach, a.id, a.attachpid, a.attachlvl, b.id, b.attachpid, b.attachlvl
from tbldoc a with(nolock)
right join tbldoc b on a.begdoc# = b.srm_begattach
where b.[path] like '%REV004%'
order by a.begdoc#

select count(*) from tbldoc with(nolock)
where [path] like '%REV006%'
