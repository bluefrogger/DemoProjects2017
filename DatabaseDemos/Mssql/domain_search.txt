--Based on Parent Emails
select count(*) from tbldoc with(nolock)
where (([to] like '%Columbia%' or [from] like '%Columbia%' or cc like '%Columbia%' or bcc like '%Columbia%')
and id = attachpid)
or 
(([to] like '%Columbia%' or [from] like '%Columbia%' or cc like '%Columbia%' or bcc like '%Columbia%')
and attachpid = 0)
go

select count(*) from tbldoc with(nolock)
where (([to] like '%Columbia%' or [from] like '%Columbia%' or cc like '%Columbia%' or bcc like '%Columbia%')
and
([to] like '%PENN%' or [from] like '%PENN%' or cc like '%PENN%' or bcc like '%PENN%')
and
id = attachpid)
or
(([to] like '%Columbia%' or [from] like '%Columbia%' or cc like '%Columbia%' or bcc like '%Columbia%')
and
([to] like '%PENN%' or [from] like '%PENN%' or cc like '%PENN%' or bcc like '%PENN%')
and
attachpid = 0)
go
