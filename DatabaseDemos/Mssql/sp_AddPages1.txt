ALTER PROCEDURE [dbo].[sp_AddPages1]
@ID int,
@Pages int,
@OcrFlag char(1)
AS
BEGIN
DECLARE @I int
SET @I = 1
WHILE @I <=  @Pages
BEGIN
   insert into tblpage (id,page,header,footer,ocrflag) values (@ID,@I,0,0,@OcrFlag)
   SET @I = @I + 1
END
END
